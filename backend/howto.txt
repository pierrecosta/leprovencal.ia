Installe PostgreSQL 
	sudo apt update
	sudo apt install libpq-dev gcc python3-dev
	sudo apt install postgresql postgresql-contrib

Initialisation de la base
	sudo -u postgres psql
		CREATE USER myuser WITH PASSWORD 'mypassword';
		CREATE DATABASE provencal_db OWNER myuser;
		GRANT ALL PRIVILEGES ON DATABASE provencal_db TO myuser;
		\q

Initialisation de l'environnement
	python -m venv venv
	source venv/bin/activate
	python -m pip install -r requirements.txt

Variables d'environnement (.env)
	DATABASE_URL=postgresql+psycopg2://myuser:mypassword@localhost:5432/provencal_db
	# Optionnel (CORS) - liste séparée par des virgules
	ALLOWED_ORIGINS=http://localhost:3000,http://127.0.0.1:3000
